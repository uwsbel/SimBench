import pychrono as chrono
import pychrono.irrlicht as chronoirr
import math

class MyEmitter(chrono.ChParticleEmitter):
    def __init__(self, mphysical_system):
        super().__init__(mphysical_system)
        self.m_num_particles = 10
        self.m_particle_radius = 0.5
        self.m_particle_mass = 1.0
        self.m_particle_color = chrono.ChColor(0.6, 0.6, 0.6)
        self.m_particle_texture = ""
        self.m_emit_direction = chrono.ChVector3d(0, 1, 0)
        self.m_emit_origin = chrono.ChVector3d(0, 0, 0)
        self.m_emit_speed = 0
        self.m_emit_angle = 0
        self.m_emit_time_interval = 0.1
        self.m_particle_system = chrono.ChParticleSystem()
        self.m_particle_system.SetGravity(chrono.ChVector3d(0, 0, 0))
        self.m_particle_system.SetUseParticleRotations(True)
        self.m_particle_system.SetUseParticleTextures(True)
        self.m_particle_system.SetUseParticleColors(True)
        self.m_particle_system.SetUseParticleSpheres(True)
        self.m_particle_system.SetSphericalParticles(True)

        # Create a ground body
        ground = chrono.ChBodyEasyBox(10, 1, 10, 1000, True, True)
        ground.SetPos(chrono.ChVector3d(0, -1, 0))
        ground.SetFixed(True)
        ground.GetVisualShape(0).SetTexture(chrono.GetChronoDataFile("textures/concrete.jpg"))
        self.mphysical_system.Add(ground)

    def EmitParticles(self):
        for i in range(self.m_num_particles):
            particle = chrono.ChParticle()
            particle.SetMass(self.m_particle_mass)
            particle.SetRadius(self.m_particle_radius)
            particle.SetColor(self.m_particle_color)
            if self.m_particle_texture != "":
                particle.SetTexture(self.m_particle_texture)
            self.AddParticle(particle)

            # Randomly set the initial position, velocity, and orientation of the particle
            rand_pos = chrono.ChFrand() * 2 * self.m_particle_radius
            particle.SetPos(self.m_emit_origin + chrono.ChVector3d(rand_pos, 0, chrono.ChFrand() * 2 * self.m_particle_radius))
            rand_vel = chrono.ChFrand() * 2 * self.m_particle_radius
            particle.SetPosDt(chrono.ChVector3d(rand_vel, self.m_emit_speed, chrono.ChFrand() * 2 * self.m_particle_radius))
            rand_rot = chrono.ChFrand() * 2 * math.pi
            particle.SetRot(chrono.ChQuaterniond(math.cos(rand_rot / 2), 0, math.sin(rand_rot / 2), 0))
            particle.SetRotDt(chrono.ChVector3d(0, 0, 0))

            self.m_particle_system.AddParticle(particle)

        self.m_num_particles = 0

    def SetParticleSystem(self, mparticle_system):
        self.m_particle_system = mparticle_system

class MyForce(chrono.ChForce):
    def __init__(self):
        super().__init__()

    def Update(self, time):
        self.ClearAllAccumulators()
        self.SetF_x(0)
        self.SetF_y(0)
        self.SetF_z(0)

        # Apply custom gravitational force
        G = 6.674 * 1e-11  # Gravitational constant
        for p in self.GetOwnerBody().GetSystem().Get_list_bodies():
            if p is not self.GetOwnerBody():
                r = p.GetPos() - self.GetOwnerBody().GetPos()
                d = r.Length()
                if d > 0:
                    F = G * (self.GetOwnerBody().GetMass() * p.GetMass()) / (d * d)
                    self.SetF_x(self.GetF_x() + F * (r.x / d))
                    self.SetF_y(self.GetF_y() + F * (r.y / d))
                    self.SetF_z(self.GetF_z() + F * (r.z / d))
